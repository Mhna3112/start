a = int(input())## nhap so luong phan tu 
so_chan =0
for i in range(1,a+1): ## cho i chay tu 1-N
    c = float(input(f"nhap so {i}: ")) ##nhap so vao danh sach
    if c%2==0: #chia het cho 2 thi lay
        so_chan +=1 ##them so chan vo danh sach 
        
print(f"so chan trong danh sach : {so_chan}")
