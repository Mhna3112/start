a = int(input())## nhap so luong phan tu 
ds = []
for i in range(1,a+1): ## cho i chay tu 1-N
    c = float(input(f"nhap so {i}: ")) ##nhap so vao danh sach
    ds.append(c)
    
so_nho_nhat = min(ds)
print(f"gia tri nho nhat {so_nho_nhat}")
so_lon_nhat = max(ds)
print(f"gia tri lon nhat {so_lon_nhat}")
